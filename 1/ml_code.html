#第二次预测,猜测性别对存活的影响
#将性别作为因子,进行绘图查看
sns.factorplot(x="Survived",col="Sex",kind="count",data=df_train)
#从下图可以看到,女性的存活率比男性的存活率高,所以可以假设所有男性全部死亡,全部女性存活
df_test['Survived']=df_test.Sex=="female"
df_test['Survived']=df_test.Survived.apply(lambda x:int(x))
df_test.head()
df_test[["PassengerId","Survived"]].to_csv("./data/predictions/female_survived.csv",index=False)
#提交到Kaggle,结果为:0.76555
